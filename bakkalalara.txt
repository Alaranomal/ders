create table müsterilerr(
müsteri_id int primary key,
müsteri_ad varchar(40),
müsteri_soyad varchar(40),
müsteri_telefon int)

satis_id int primary key,
tarih datetime,
toplam_tutar int,
gider_id int,
müsteri_id int)

kategori_id int primary key,
kategori_adi varchar(40))

create table firmaa(
firma_id int primary key,
firma_adi varchar(40),
alinan_urun_id int,
tarih datetime);

create table ürünlerr(
urun_id int primary key,
urun_adi varchar(40),
kategori_id int,
firma_id int,
stok_adet int,
alis_fiyat int,
satis_fiyat int)

create table urun_detayy(
urun_id int, 
urun_detay_id int primary key,
satis_id int,
adet int)

create table giderlerr(
gider_id int primary key,
gider_tipi varchar(40),
gider_tutari varchar(40),
gider_tarihi datetime)

create table satis_detayy(
satis_id int,
satis_detay_id int primary key,
urun_id int,
miktar int,
birim_fiyat int,
tutar int)
